function titleCase(str) {
   var ret = "";
   var upperCase = true;
   for (var i = 0; i < str.length; i++) {
    var chr = str.charAt(i);
    if(upperCase) {
      ret += chr.toUpperCase();
    }
    else{
      ret += chr.toLowerCase();
    }
    upperCase = chr == " ";
  }
  return ret;
}
titleCase("I'm a little tea pot");
titleCase("sHoRt AnD sToUt");
titleCase("HERE IS MY HANDLE HERE IS MY SPOUT");
